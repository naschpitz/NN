This page provides practical examples for using the CNN library. Each example demonstrates a key feature with complete, runnable code.

1. Creating a CNN

Configure a convolutional neural network with CNN layers followed by dense layers:

#include "CNN_Core.hpp"

int main() {
    // Configure CNN layers
    CNN::LayersConfig layers;
    
    // Conv(8 filters, 3×3, stride 1, VALID) → ReLU → MaxPool(2×2, stride 2)
    layers.cnnLayers = {
        {CNN::LayerType::CONV, CNN::ConvLayerConfig{
            8,                                    // numFilters
            3, 3,                              // filterH, filterW
            1, 1,                              // strideY, strideX
            CNN::SlidingStrategyType::VALID
        }},
        {CNN::LayerType::RELU, CNN::ReLULayerConfig{}},
        {CNN::LayerType::POOL, CNN::PoolLayerConfig{
            CNN::PoolTypeEnum::MAX, 2, 2, 2, 2
        }},
        {CNN::LayerType::FLATTEN, CNN::FlattenLayerConfig{}}
    };
    
    // Configure dense layers: 64 RELU → 10 SIGMOID
    layers.denseLayers = {
        {64, ANN::ActvFuncType::RELU},
        {10, ANN::ActvFuncType::SIGMOID}
    };
    
    // Create core config
    CNN::CoreConfig<float> config;
    config.deviceType = CNN::DeviceType::CPU;
    config.modeType = CNN::ModeType::TRAIN;
    config.inputShape = {1, 28, 28};  // 1-channel 28×28 image
    config.layersConfig = layers;
    config.trainingConfig.numEpochs = 10;
    config.trainingConfig.batchSize = 64;
    config.trainingConfig.learningRate = 0.01f;
    
    // Create the core
    auto core = CNN::Core<float>::makeCore(config);
    
    return 0;
}
The inputShape is specified as {channels, height, width}. For MNIST-like datasets, use {1, 28, 28}. For RGB images, use {3, height, width}.

2. Training with Samples

Create training samples and train the network:

// Create sample data
CNN::Samples<float> samples;

// Build a few sample entries
for (int i = 0; i < 100; ++i) {
    CNN::Sample<float> sample;
    
    // Create input: 1×28×28 tensor
    sample.input = CNN::Input<float>({1, 28, 28});
    // Fill with random data (or load from file)
    for (auto& val : sample.input.data) {
        val = static_cast<float>(rand()) / RAND_MAX;
    }
    
    // Create output: 10-element vector (one-hot encoded)
    sample.output = CNN::Output<float>(10, 0.0f);
    sample.output[i % 10] = 1.0f;  // Set correct class
    
    samples.push_back(sample);
}

// Configure training
config.trainingConfig.numEpochs = 10;
config.trainingConfig.batchSize = 64;
config.trainingConfig.learningRate = 0.01f;
config.progressReports = 1000;  // Progress frequency (all modes)

// Train the network
core->train(samples);

// Access training metadata
auto metadata = core->trainingMetadata;
std::cout << "Training duration: " << metadata.durationFormatted << std::endl;
std::cout << "Final loss: " << metadata.finalLoss << std::endl;

3. Prediction

Perform inference on a single input:

// Create input tensor with proper shape
CNN::Input<float> input({1, 28, 28});
// Fill with data...
for (auto& val : input.data) {
    val = static_cast<float>(rand()) / RAND_MAX;
}

// Get prediction
CNN::Output<float> output = core->predict(input);

// Print results (10 output neurons)
for (size_t i = 0; i < output.size(); ++i) {
    std::cout << "Class " << i << ": " << output[i] << std::endl;
}

4. Testing

Evaluate the network on a test set:

// Create test samples (same format as training)
CNN::Samples<float> testSamples;
// ... populate testSamples ...

// Run test
CNN::TestResult<float> result = core->test(testSamples);

// Print results
std::cout << "Test samples: " << result.numSamples << std::endl;
std::cout << "Total loss: " << result.totalLoss << std::endl;
std::cout << "Average loss: " << result.averageLoss << std::endl;

5. GPU Training

Switch to GPU acceleration with minimal code changes:

// Same configuration as CPU, but with GPU device type
CNN::CoreConfig<float> config;
config.deviceType = CNN::DeviceType::GPU;  // Switch to GPU
config.modeType = CNN::ModeType::TRAIN;
config.inputShape = {1, 28, 28};
config.layersConfig = layers;

// Configure GPU training
config.numGPUs = 0;  // 0 = use all available GPUs
config.trainingConfig.numEpochs = 10;
config.trainingConfig.learningRate = 0.01f;

// Create and train (everything else is identical)
auto core = CNN::Core<float>::makeCore(config);
core->train(samples);
Set numGPUs = 0 to automatically use all available GPUs, or specify a number to use that many GPUs. GPU training requires OpenCL 1.2 or later.

6. Step-by-Step Training

Manual epoch loop with custom callbacks:

// Configure for manual control
config.trainingConfig.numEpochs = 1;  // Train one epoch at a time

for (int epoch = 0; epoch < 10; ++epoch) {
    // Train one epoch
    core->train(samples);
    
    // Access training metadata after each epoch
    auto metadata = core->trainingMetadata;
    std::cout << "Epoch " << epoch << ": loss = " 
              << metadata.finalLoss << std::endl;
    
    // Optionally test on validation set
    auto testResult = core->test(valSamples);
    std::cout << "  Val loss: " << testResult.averageLoss << std::endl;
}
